local function parseValue(text)
    if not text then return 0 end
    text = tostring(text)

    -- ignora READY
    if text:find("READY") then return 0 end

    -- ignora timer
    if text:find("h") or text:find("m") then return 0 end

    -- aceita só B M K
    if not text:find("B") and not text:find("M") and not text:find("K") then
        return 0
    end

    local clean = text:gsub("[^%d%.BMK]", "")

    local mult = 1
    if clean:find("B") then
        mult = 1e9
    elseif clean:find("M") then
        mult = 1e6
    elseif clean:find("K") then
        mult = 1e3
    end

    local num = clean:gsub("[BMK]", "")
    return (tonumber(num) or 0) * mult
end
